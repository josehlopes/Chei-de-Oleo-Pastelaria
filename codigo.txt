CREATE DATABASE IF NOT EXISTS Chei_de_oleo_pastelaria;

USE Chei_de_oleo_pastelaria;

CREATE TABLE IF NOT EXISTS clientes (
    idCliente INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
    nomeCliente VARCHAR(100) NOT NULL,
    INDEX (nomeCliente)
);

CREATE TABLE IF NOT EXISTS enderecos (
    idEndereco INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
    logradouro VARCHAR(100) NOT NULL,
    numero CHAR(10) NOT NULL,
    cep CHAR(8) NOT NULL,
    complemento VARCHAR(100),
    idCliente INT NOT NULL,
    FOREIGN KEY (idCliente) REFERENCES clientes (idCliente),
    INDEX (logradouro)
);

CREATE TABLE IF NOT EXISTS contatos (
    idContato INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
    telefone1 CHAR(11) NOT NULL,
    telefone2 CHAR(11),
    idCliente INT NOT NULL,
    FOREIGN KEY (idCliente) REFERENCES clientes (idCliente),
    INDEX (telefone1),
    INDEX (telefone2)
);

CREATE TABLE IF NOT EXISTS ingredientes (
    idIngrediente INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
    nome VARCHAR(100) NOT NULL,
    INDEX (nome)
);

CREATE TABLE IF NOT EXISTS pasteis (
    idPastel INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
    descricao VARCHAR(100),
    tamanho ENUM('P', 'M', 'G') NOT NULL DEFAULT 'P',
    preco DECIMAL(10, 2) NOT NULL,
    INDEX (descricao),
    INDEX (tamanho)
);

CREATE TABLE IF NOT EXISTS ingredientes_pastel (
    idPastel INT NOT NULL,
    idIngrediente INT NOT NULL,
    UNIQUE (idPastel, idIngrediente),
    FOREIGN KEY (idPastel) REFERENCES pasteis (idPastel),
    FOREIGN KEY (idIngrediente) REFERENCES ingredientes (idIngrediente)
);

CREATE TABLE IF NOT EXISTS bebidas (
    idBebida INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
    nome VARCHAR(100),
    tamanho ENUM('P', 'M', 'G') NOT NULL DEFAULT 'P',
    preco DECIMAL(10, 2) NOT NULL,
    INDEX (nome),
    INDEX (tamanho)
);

CREATE TABLE IF NOT EXISTS produtos (
    idProduto INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
    nome VARCHAR(100),
    preco DECIMAL(10, 2) NOT NULL,
    idPastel INT,
    idBebida INT,
    FOREIGN KEY (idPastel) REFERENCES pasteis (idPastel),
    FOREIGN KEY (idBebida) REFERENCES bebidas (idBebida),
    INDEX (nome)
);

CREATE TABLE IF NOT EXISTS categorias (
	idCategoria INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
    nome VARCHAR(10) NOT NULL DEFAULT 'Comum',
    idProduto INT NOT NULL,
    FOREIGN KEY (idProduto) REFERENCES produtos (idProduto),
    INDEX (nome)
);

CREATE TABLE IF NOT EXISTS formas_pagamento (
    idFormaPagamento INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
    descricao VARCHAR(100) NOT NULL,
    INDEX (descricao)
);

CREATE TABLE IF NOT EXISTS pedidos (
    idPedido INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
    idCliente INT NOT NULL,
    formaPagamento INT,
    status ENUM('Pedido realizado', 'Preparando', 'Pronto para retirada', 'Entregue') NOT NULL DEFAULT 'Pedido realizado',
    dataPedido DATE NOT NULL DEFAULT (CURRENT_DATE),
    FOREIGN KEY (idCliente) REFERENCES clientes (idCliente),
    FOREIGN KEY (formaPagamento) REFERENCES formas_pagamento (idFormaPagamento),
    INDEX (dataPedido),
    INDEX (status)
);

CREATE TABLE IF NOT EXISTS detalhes_pedido (
    idDetalhe INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
    idPedido INT NOT NULL,
    idProduto INT NOT NULL,
    quantidade INT NOT NULL,
    FOREIGN KEY (idPedido) REFERENCES pedidos (idPedido),
    FOREIGN KEY (idProduto) REFERENCES produtos (idProduto),
    INDEX (quantidade)
);
